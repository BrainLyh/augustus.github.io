  $res = '';

    foreach($obj as $key => $val) {

          foreach($val as $v) {
            $res .= "<$key>";
            $res .= json2xml($v);
            $res .= "</$key>";

          $res .= json2xml($val);
          $res .= "</$key>";

function verify($data, $hmac) {
    $secret = $_SESSION['secret'];
    if (empty($secret)) return false;
    return hash_equals(hash_hmac('sha256', $data, $secret), $hmac);

function hmac($data) {
    $secret = $_SESSION['secret'];
    if (empty($data) || empty($secret)) return false;
